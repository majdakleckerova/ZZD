Úkol 3

Spočítejte celkový počet seriálů a filmů v různých letech dle data vydání, tj. kolik celkem bylo filmů/seriálů natočeno do daného roku.

rok_vydani <- data$release_year
typ <- data$type
df <- data.frame(typ, rok_vydani)
filmy <- df %>% filter(str_detect("Movie", typ))
serialy <- df %>% filter(str_detect("TV Show",typ))
cetnost_filmy <- filmy %>% group_by(rok_vydani) %>%
  summarise(n=n())
cetnost_serialy <- serialy %>% group_by(rok_vydani) %>%
  summarise(n=n())
cs_filmy = cumsum(cetnost_filmy$n)
cs_serialy = cumsum(cetnost_serialy$n)

ggplot() +
  geom_line(data = cetnost_filmy, aes(x = rok_vydani, y = cs_filmy, color = "Filmy"), size = 0.5) +
  geom_line(data = cetnost_serialy, aes(x = rok_vydani, y = cs_serialy, color = "Seriály"), size = 0.5) +
  geom_point(data=cetnost_filmy, aes(x=rok_vydani,y=cs_filmy,colour = "Filmy"), size = 1) +
  geom_point(data=cetnost_serialy, aes(x=rok_vydani, y=cs_serialy, colour="Seriály"), size = 1) +
  labs(x = "Roky", y = "Počet", color = "Typ") +  
  scale_color_manual(values = c("Filmy" = "deeppink", "Seriály" = "darkviolet"))
